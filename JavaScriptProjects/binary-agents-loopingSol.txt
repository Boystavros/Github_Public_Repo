Binary Agents (initial attempt)
=============
Return an English translated sentence of the passed binary string.

The binary string will be space separated.
===========================================================================================================================

function binaryAgent(str) {
  let splitBins = str.split(' ');
  let letters = [];
  for(let i = 0; i < splitBins.length; i++){
    letters.push(String.fromCharCode(parseInt(splitBins[i],2)));
  }
  return letters.join("");
}
